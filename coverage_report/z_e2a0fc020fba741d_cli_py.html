384def build_fleet_report(db: Path, top_n: int = 5, recent_seconds: int = 86400, module_min_hosts: int = 3) -> dict: 

385 """Return fleet report object (not written to disk).""" 

386 store = BaselineStore(db) 

387 # 1. Collect latest finding.count.total per host 

388 latest = store.latest_metric_values('finding.count.total') 

389 host_values = [v for (_,v,_) in latest] 

390 import math 

391 mean = sum(host_values)/len(host_values) if host_values else 0.0 

392 var = sum((v-mean)**2 for v in host_values)/(len(host_values)-1) if len(host_values) > 1 else 0.0 

393 std = math.sqrt(var) 

394 hosts_stats = [] 

395 for host_id, value, ts in latest: 

396 z = (value-mean)/std if std else 0.0 

397 hosts_stats.append({"host_id": host_id, "value": value, "z": z, "ts": ts}) 

398 top_outlier_hosts = sorted(hosts_stats, key=lambda x: abs(x['z']), reverse=True)[:top_n] 

399 # 2. Newly common modules: modules whose first_seen across >= module_min_hosts hosts within recent_seconds 

400 recent_map = store.recent_module_first_seen(within_seconds=recent_seconds) 

401 newly_common_modules = [ 

402 {"module": m, "host_count": len(hs), "hosts": hs} 

403 for m, hs in recent_map.items() if len(hs) >= module_min_hosts 

404 ] 

405 # 3. Risk distribution histogram: we approximate using latest risk.sum.medium_high metric per host (if present) 

406 risk_latest = store.latest_metric_values('risk.sum.medium_high') 

407 risk_values = [v for (_,v,_) in risk_latest] 

408 # Bucket edges (log-ish or simple linear). We'll use simple linear 0-50,50-100,... up to 500 

409 buckets = list(range(0, 501, 50)) 

410 histogram = [] 

411 for i in range(len(buckets)-1): 

412 lo, hi = buckets[i], buckets[i+1] 

413 cnt = sum(1 for v in risk_values if lo <= v < hi) 

414 histogram.append({"range": f"{lo}-{hi-1}", "count": cnt}) 

415 if risk_values: 

416 cnt = sum(1 for v in risk_values if v >= buckets[-1]) 

417 histogram.append({"range": f">={buckets[-1]}", "count": cnt}) 

418 return { 

419 "generated_ts": int(__import__('time').time()), 

420 "host_count": len(host_values), 

421 "metric_mean": mean, 

422 "metric_std": std, 

423 "top_outlier_hosts": top_outlier_hosts, 

424 "newly_common_modules": newly_common_modules, 

425 "risk_distribution": histogram 

426 }
