29def render(output: EnrichedOutput) -> str: 

30 now = datetime.datetime.utcnow().isoformat() + 'Z' 

31 findings = output.enriched_findings or [] 

32 corrs = output.correlations or [] 

33 metrics = (output.summaries.metrics if output.summaries else {}) or {} 

34 # Compliance summary table & gaps rendering 

35 comp = metrics.get('compliance_summary') if isinstance(metrics, dict) else None 

36 comp_rows = '' 

37 if comp: 

38 comp_rows = '<table class="table"><tr><th>Standard</th><th>Passed</th><th>Failed</th><th>N/A</th><th>Total</th><th>Score</th></tr>' 

39 for std, vals in comp.items(): 

40 passed = vals.get('passed'); failed = vals.get('failed'); na = vals.get('not_applicable') or 0 

41 total = vals.get('total_controls'); score = vals.get('score') 

42 comp_rows += f"<tr><td>{html.escape(str(std))}</td><td>{passed}</td><td>{failed}</td><td>{na}</td><td>{total}</td><td>{score}</td></tr>" 

43 comp_rows += '</table>' 

44 gaps = metrics.get('compliance_gaps') if isinstance(metrics, dict) else None 

45 gap_rows = '' 

46 if gaps: 

47 gap_rows = '<table class="table"><tr><th>Standard</th><th>Control</th><th>Severity</th><th>Hint</th></tr>' 

48 for g in gaps[:30]: 

49 gap_rows += f"<tr><td>{html.escape(str(g.get('standard')))}</td><td>{html.escape(str(g.get('control_id')))}</td><td>{html.escape(str(g.get('severity') or ''))}</td><td>{html.escape(str(g.get('remediation_hint') or ''))}</td></tr>" 

50 gap_rows += '</table>' 

51 rows = [] 

52 for f in findings[:400]: # safety cap 

53 sev = (f.severity or 'info').lower() 

54 tags = ''.join(f'<span class="badge">{html.escape(t)}</span>' for t in (f.tags or [])[:12]) 

55 rationale = '' 

56 if f.rationale: 

57 rationale = '<ul>' + ''.join(f'<li>{html.escape(r)}</li>' for r in f.rationale[:6]) + '</ul>' 

58 # Format probability defensively (older models may not set it) 

59 if f.probability_actionable is not None: 

60 prob_fmt = f"{f.probability_actionable:.2f}" 

61 else: 

62 prob_fmt = "0.00" 

63 rows.append(f"<div class='finding {sev}'><strong>{html.escape(f.title or f.id)}</strong> <span class='badge sev-{sev}'>{sev}</span> risk={f.risk_total or f.risk_score} prob={prob_fmt} {tags}{rationale}</div>") 

64 corr_rows = [] 

65 for c in corrs[:120]: 

66 corr_rows.append(f"<div class='finding'><strong>{html.escape(c.title)}</strong> <small>{len(c.related_finding_ids)} findings</small><br>{html.escape(c.rationale or '')}</div>") 

67 exec_summary = html.escape(output.summaries.executive_summary if output.summaries and output.summaries.executive_summary else '(no executive summary)') 

68 attack = output.summaries.attack_coverage if output.summaries else {} 

69 attack_html = '' 

70 if attack: 

71 attack_html = '<div class="card"><h3>ATT&CK Coverage</h3>' + f"Techniques: {attack.get('technique_count')}<br>" + '</div>' 

72 return f"""<!DOCTYPE html><html><head><meta charset='utf-8'><title>sys-scan Report</title><style>{CSS}</style></head> 

73<body><header><h1>sys-scan Enriched Report</h1><small>Generated {now}</small></header> 

74<section class='summary-block'><h2>Executive Summary</h2><p>{exec_summary}</p></section> 

75<div class='flex'> 

76<div class='card'><h3>Metrics</h3><pre>{html.escape(json.dumps(metrics, indent=2)[:4000])}</pre></div> 

77{attack_html} 

78<div class='card'><h3>Compliance</h3>{comp_rows if comp_rows else '<small>No compliance data</small>'}</div> 

79<div class='card'><h3>Correlations</h3><p>{len(corrs)} correlation(s)</p></div> 

80</div> 

81{('<section class="summary-block"><h2>Compliance Gaps</h2>' + gap_rows + '</section>') if gap_rows else ''} 

82<h2>Findings ({len(findings)})</h2> 

83{''.join(rows)} 

84<h2>Correlations</h2> 

85{''.join(corr_rows)} 

86<footer>Static HTML artifact. No external JS. sys-scan.</footer> 

87</body></html>"""
