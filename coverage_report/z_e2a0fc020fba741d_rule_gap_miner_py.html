10class GapCandidate: 

11 def __init__(self, key: str): 

12 self.key = key 

13 self.count = 0 

14 self.examples: List[Dict[str,Any]] = [] 

15 self.tokens: Dict[str,int] = {} 

16 self.metadata_keys: Dict[str,int] = {}

19 def add(self, finding: Dict[str,Any]): 

20 self.count += 1 

21 title = finding.get('title','') 

22 for t in set(tok.lower() for tok in TOKEN_RE.findall(title) if len(tok) > 3): 

23 self.tokens[t] = self.tokens.get(t,0)+1 

24 meta = finding.get('metadata') or {} 

25 if isinstance(meta, dict): 

26 for k,v in meta.items(): 

27 if isinstance(v,(str,int,float)): 

28 self.metadata_keys[k] = self.metadata_keys.get(k,0)+1 

29 sev = finding.get('severity','unknown').lower() 

30 self.severities[sev] = self.severities.get(sev,0)+1 

31 if len(self.examples) < 5: 

32 self.examples.append({k:v for k,v in finding.items() if k in {'id','title','severity','metadata','tags'}})

34 def rule_skeleton(self) -> Dict[str,Any]: 

35 # Pick 1-2 distinctive tokens (highest doc freq but not overly common terms) 

36 token_items = sorted(self.tokens.items(), key=lambda x: (-x[1], x[0])) 

37 chosen_tokens = [t for t,_ in token_items[:2]] or [self.key] 

38 conditions = [] 

39 for tok in chosen_tokens: 

40 conditions.append({'field':'title','contains':tok}) 

41 # Metadata conditions (only if appear in all examples and scalar simple value) 

42 meta_always = [] 

43 if self.examples: 

44 common_keys = set(self.examples[0].get('metadata',{}).keys()) 

45 for ex in self.examples[1:]: 

46 common_keys &= set((ex.get('metadata') or {}).keys()) 

47 for k in sorted(common_keys): 

48 vals = { (ex.get('metadata') or {}).get(k) for ex in self.examples } 

49 if len(vals) == 1: 

50 v = list(vals)[0] 

51 if isinstance(v,(str,int,float)) and len(str(v)) < 64: 

52 meta_always.append({'metadata_key':k,'metadata_contains':str(v)}) 

53 conditions.extend(meta_always[:2]) 

54 rid = f"gap_{self.key[:40]}" 

55 title_tokens = ' '.join(t.capitalize() for t in chosen_tokens[:3]) 

56 return { 

57 'id': rid, 

58 'title': f"Auto Suggested: {title_tokens}", 

59 'rationale': f"Suggested from {self.count} recurring uncorrrelated high-risk findings", 

60 'conditions': conditions, 

61 'logic': 'all', 

62 'risk_score_delta': 5, 

63 'tags': ['auto_suggested','gap_candidate'] 

64 }

67def normalize_title(title: str) -> str: 

68 # Remove digits/paths, keep tokens 

69 tokens = [t.lower() for t in TOKEN_RE.findall(title) if not t.isdigit()] 

70 if not tokens: 

71 return 'untitled' 

72 return '_'.join(tokens[:6])

75def mine_gap_candidates(paths: List[Path], risk_threshold: int = 60, min_support: int = 3) -> Dict[str,Any]: 

76 candidates: Dict[str, GapCandidate] = {} 

77 for p in paths: 

78 try: 

79 data = json.loads(p.read_text()) 

80 except Exception: 

81 continue 

82 findings = data.get('enriched_findings') or [] 

83 if not isinstance(findings, list): 

84 continue 

85 for f in findings: 

86 if not isinstance(f, dict): 

87 continue 

88 sev = f.get('severity','').lower() 

89 risk = f.get('risk_total') or f.get('risk_score') or 0 

90 # skip if correlated 

91 if f.get('correlation_refs'): 

92 continue 

93 # treat high severity or risk threshold as candidate 

94 if sev not in {'high','critical'} and risk < risk_threshold: 

95 continue 

96 # Skip if any existing rule tag present 

97 tags = [t for t in f.get('tags') or [] if isinstance(t,str)] 

98 if any(t.startswith('rule:') for t in tags): 

99 continue 

100 key = normalize_title(f.get('title','')) 

101 cand = candidates.get(key) 

102 if not cand: 

103 cand = GapCandidate(key) 

104 candidates[key] = cand 

105 cand.add(f) 

106 # Filter support 

107 selected = [c for c in candidates.values() if c.count >= min_support] 

108 selected.sort(key=lambda c: c.count, reverse=True) 

109 suggestions = [c.rule_skeleton() for c in selected] 

110 return { 

111 'total_candidates': len(candidates), 

112 'selected': len(selected), 

113 'suggestions': suggestions, 

114 'candidates': [ {'key':c.key,'count':c.count,'example_titles':[e['title'] for e in c.examples]} for c in selected ] 

115 }

118def refine_with_llm(suggestions: List[Dict[str,Any]], examples: Optional[Dict[str,List[str]]] = None) -> List[Dict[str,Any]]: 

119 """Augment rule skeletons with improved rationale & token refinement using LLMClient heuristics. 

120 Since we lack an external LLM call here, leverage deterministic heuristics: 

121 - Concatenate example titles and extract most common tokens not already in conditions. 

122 - Expand rationale with token list. 

123 - Add 'refined' tag. 

124 """ 

125 client = LLMClient() # placeholder (not actually used for generation now) 

126 refined = [] 

127 for s in suggestions: 

128 rid_val = s.get('id') or '' 

129 ex_titles = examples.get(rid_val, []) if examples else [] 

130 token_freq: Dict[str,int] = {} 

131 for t in ex_titles: 

132 for tok in TOKEN_RE.findall(t.lower()): 

133 if len(tok) < 4: continue 

134 token_freq[tok] = token_freq.get(tok,0)+1 

135 existing_tokens = {c.get('contains') for c in (s.get('conditions') or []) if 'contains' in c} 

136 extra = [tok for tok,_ in sorted(token_freq.items(), key=lambda x:(-x[1], x[0])) if tok not in existing_tokens][:3] 

137 if extra: 

138 # Add one additional condition (any logic) if not duplicate 

139 for e in extra: 

140 s.setdefault('conditions', []).append({'field':'title','contains':e}) 

141 # Switch logic to any if multiple condition families present 

142 if len(s['conditions']) > 3: 

143 s['logic'] = 'any' 

144 rationale = s.get('rationale','') 

145 if extra: 

146 s['rationale'] = rationale + f" | refined tokens: {', '.join(extra)}" 

147 tags = s.get('tags',[]) 

148 if 'refined' not in tags: 

149 tags.append('refined') 

150 s['tags'] = tags 

151 refined.append(s) 

152 # Optional secondary LLM refinement layer (real model integration stub) 

153 if os.environ.get('AGENT_RULE_REFINER_USE_LLM') == '1': 

154 try: 

155 from .rule_refiner import llm_refine 

156 refined = llm_refine(refined, examples or {}) 

157 except Exception: 

158 pass 

159 return refined
